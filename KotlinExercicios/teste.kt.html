fun calcularPerdaDeVida() {
    println("Quantos cigarros você fuma por dia:")
    val quantidadeCigarros = readln().toInt()
    println("Por quantos anos você já fumou:")
    val anosFumando = readln().toInt()

    val minutosPerdidosPorDia = quantidadeCigarros * 10
    val horasPerdidasPorDia = minutosPerdidosPorDia / 60.0
    val diasPerdidosPorAno = horasPerdidasPorDia / 24
    val diasPerdidosTotal = diasPerdidosPorAno * anosFumando

    println("O fumante perdeu aproximadamente %.2f dias de vida.".format(diasPerdidosTotal))
}

fun verificarTriangulo() {
    println("Digite os três lados do triângulo:")
    val segmentoA = readln().toDouble()
    val segmentoB = readln().toDouble()
    val segmentoC = readln().toDouble()

    if (segmentoA <= 0 || segmentoB <= 0 || segmentoC <= 0) {
        println("Os lados do triângulo devem ser valores positivos.")
        return
    }

    if (segmentoA < segmentoB + segmentoC && segmentoB < segmentoA + segmentoC && segmentoC < segmentoA + segmentoB) {
        println("É possível formar um triângulo com esses segmentos de reta.")
    } else {
        println("Não é possível formar um triângulo com esses segmentos de reta.")
    }
}

fun classificarTriangulo() {
    println("Digite os três lados do triângulo:")
    val ladoA = readln().toDouble()
    val ladoB = readln().toDouble()
    val ladoC = readln().toDouble()

    if (ladoA < ladoB + ladoC && ladoB < ladoA + ladoC && ladoC < ladoA + ladoB) {
        println("É possível formar um triângulo com esses segmentos de reta.")
        when {
            ladoA == ladoB && ladoB == ladoC -> println("O triângulo é EQUILÁTERO.")
            ladoA == ladoB || ladoB == ladoC || ladoA == ladoC -> println("O triângulo é ISÓSCELES.")
            else -> println("O triângulo é ESCALENO.")
        }
    } else {
        println("Não é possível formar um triângulo com esses segmentos de reta.")
    }
}

    when {
        escolhaJogador == escolhaComputador -> println("Empate!")
        (escolhaJogador == 1 && escolhaComputador == 3) ||

fun adivinheONumero() {
    println("🎉 Bem-vindo ao jogo de adivinhação!")
    println("🔢 Tente adivinhar o número sorteado entre 1 e 5.")

    val numeroSorteado = Random.nextInt(1, 6)
    var tentativas = 0

    do {
        print("📝 Digite seu palpite (1 a 5): ")
        val entrada = readln()

        // Verifica se a entrada é um número válido
        val numeroJogador = entrada.toIntOrNull()

        if (numeroJogador == null || numeroJogador !in 1..5) {
            println("❌ Entrada inválida! Digite um número entre 1 e 5.")
            continue
        }

        tentativas++

        when {
            numeroJogador < numeroSorteado -> println("🔼 O número sorteado é maior. Tente novamente!")
            numeroJogador > numeroSorteado -> println("🔽 O número sorteado é menor. Tente novamente!")
            else -> println("🎉 Parabéns! Você acertou o número sorteado ($numeroSorteado) em $tentativas tentativas.")
        }

    } while (numeroJogador != numeroSorteado)
}

fun adivinhaNumeroLimitado() {
    println("🎉 Bem-vindo ao jogo de adivinhação!")
    println("🔢 Tente adivinhar o número sorteado entre 1 e 10.")
    println("🎯 Você tem 4 tentativas.")

    val numeroSorteado = Random.nextInt(1, 11)
    val maxTentativas = 4
    var tentativas = 0

    while (tentativas < maxTentativas) {
        print("📝 Tentativa ${tentativas + 1} de $maxTentativas. Digite seu palpite (1 a 10): ")
        val entrada = readln()

        val numeroJogador = entrada.toIntOrNull()

        if (numeroJogador == null || numeroJogador !in 1..10) {
            println("❌ Número inválido! Digite um número entre 1 e 10.")
            continue
        }

        tentativas++

        when {
            numeroJogador == numeroSorteado -> {
                println("🎉 Parabéns! Você acertou o número sorteado ($numeroSorteado) em $tentativas tentativas. 🏆")
                return
            }
            numeroJogador < numeroSorteado -> println("🔼 O número sorteado é maior. Tente novamente!")
            else -> println("🔽 O número sorteado é menor. Tente novamente!")
        }
    }

    println("❌ Você não acertou! O número sorteado era $numeroSorteado. 😢")
}

fun pa() {
    println("📏 Cálculo dos 10 primeiros termos da Progressão Aritmética (PA)")

    print("📝 Digite o primeiro termo da PA: ")
    val primeiroTermo = readln().toInt()
    print("📈 Digite a razão da PA: ")
    val razao = readln().toInt()

    var termoAtual = primeiroTermo
    var soma = 0

    println("\n📊 Os 10 primeiros termos da PA são:")
    for (i in 1..10) {
        print("$termoAtual ➝ ")
        soma += termoAtual
        termoAtual += razao
    }
    println("FIM")

    println("\n🧮 A soma dos 10 primeiros termos da PA é: $soma")
}

fun fibonacci() {
    println("🐚 Os 10 primeiros elementos da sequência de Fibonacci são:")

    var a = 1
    var b = 1

    print("✨ $a ➝ $b ")

    for (i in 3..10) {
        val c = a + b
        print("➝ $c ")
        a = b
        b = c
    }
    println("➝ FIM")
}

fun vetorOrdenado() {
    println("🔢 Gerando um vetor de 20 números aleatórios...")

    val vetor = IntArray(20)

    for (i in vetor.indices) {
        vetor[i] = Random.nextInt(0, 100)
    }

    println("\n🎲 Números aleatórios gerados:")
    println(vetor.joinToString(" ➝ "))

    vetor.sort()

    println("\n📈 Números ordenados em ordem crescente:")
    println(vetor.joinToString(" ➝ "))
}

fun fibonacciParaLimitado(quantidade: Int) {
    if (quantidade <= 0) {
        println("⚠️ Quantidade inválida! Insira um número maior que zero.")
        return
    }

    var a = 1
    var b = 1

    print("\n🌀 Sequência de Fibonacci:\n")

    if (quantidade >= 1) print("✨ $a ➝ ")
    if (quantidade >= 2) print("$b ➝ ")

    for (i in 3..quantidade) {
        val c = a + b
        print("$c ➝ ")
        a = b
        b = c
    }

    println("FIM")
}
